262  void operator()(int64_t k, int64_t *indices, int64_t indices_length, int64_t *centers, int64_t &centers_length) {
263  UniqueRandom r(indices_length);
264 
265  int64_t index;
266  for (index = 0; index < k; index++) {
267  bool duplicate = true;
268  int64_t rnd;
269  while (duplicate) {
270  duplicate = false;
271  rnd = r.next();
272  if (rnd < 0) {
273  centers_length = index;
274  return;
275  }
276 
277  centers[index] = indices[rnd];
278  for (int j = 0; j < index; j++) {
279  auto a = points->slice(0, centers[index], centers[index] + 1).contiguous().data_ptr<float>();
280  auto b = points->slice(0, centers[j], centers[j] + 1).contiguous().data_ptr<float>();
281  auto sq = faiss::fvec_L2sqr(a, b, vecDim);
282  if (sq < 1e-16) {
283  duplicate = true;
284  }
285  }
286  }
287  }
288  centers_length = index;
289  }
