160  size_t getCurrentMemoryUsage() {
161  // Read the VmRSS field from /proc/self/status
162  std::ifstream statusFile("/proc/self/status");
163  std::string line;
164 
165  while (std::getline(statusFile, line)) {
166  if (line.compare(0, 6, "VmRSS:") == 0) {
167  size_t memoryUsageKB = std::stoul(line.substr(7));
168  return memoryUsageKB;
169  }
170  }
171 
172  return 0; // Return 0 if VmRSS is not found (error handling)

175  double get_core_utilization(int core_number) {
176  // Open /proc/stat file
177  std::ifstream stat_file("/proc/stat");
178  if (!stat_file.is_open()) {
179  std::cerr << "Error opening /proc/stat" << std::endl;
180  return -1.0;
181  }
182 
183  // Read the lines until finding the line related to the specified core
184  std::string line;
185  std::string core_label = "cpu" + std::to_string(core_number);
186  while (std::getline(stat_file, line)) {
187  if (line.compare(0, core_label.size(), core_label) == 0) {
188  break;
189  }
190  }
191 
192  // Close the file
193  stat_file.close();
194 
195  // Parse the CPU utilization values for the specified core
196  std::istringstream iss(line);
197  std::string cpuLabel;
198  long user, nice, system, idle, iowait, irq, softirq, steal, guest, guest_nice;
199  iss >> cpuLabel >> user >> nice >> system >> idle >> iowait >> irq >> softirq >> steal >> guest >> guest_nice;
200 
201  // Calculate total CPU time for the specified core
202  long totalCpuTime = user + nice + system + idle + iowait + irq + softirq + steal + guest + guest_nice;
203 
204  // Calculate the CPU utilization percentage for the specified core
205  double cpuUtilization = 100.0
206  * (1.0 - static_cast<double>(idle - totalIdleOld[core_number]) / (totalCpuTime - totalTimeOld[core_number]));
207  totalIdleOld[core_number] = idle;
208  totalTimeOld[core_number] = totalCpuTime;
209  if (cpuUtilization >= 100.0) {
210  cpuUtilization = 100.0;
211  }
212  return cpuUtilization;
213  }
