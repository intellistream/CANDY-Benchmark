166  blocks(const T1 first_index_, const T2 index_after_last_,
167  const size_t num_blocks_)
168  : first_index(static_cast<T>(first_index_)),
169  index_after_last(static_cast<T>(index_after_last_)),
170  num_blocks(num_blocks_) {
171  if (index_after_last < first_index)
172  std::swap(index_after_last, first_index);
173  total_size = static_cast<size_t>(index_after_last - first_index);
174  block_size = static_cast<size_t>(total_size / num_blocks);
175  if (block_size == 0) {
