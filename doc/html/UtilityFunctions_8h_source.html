81  static double getLatencyPercentage(double fraction, std::vector<INTELLI::IntelliTimeStampPtr> &myTs) {
82  size_t rLen = myTs.size();
83  size_t nonZeroCnt = 0;
84  std::vector<uint64_t> validLatency;
85  for (size_t i = 0; i < rLen; i++) {
86  if (myTs[i]->processedTime >= myTs[i]->arrivalTime && myTs[i]->processedTime != 0) {
87  validLatency.push_back(myTs[i]->processedTime - myTs[i]->arrivalTime);
88  nonZeroCnt++;
89  }
90  }
91  if (nonZeroCnt == 0) {
92  INTELLI_ERROR("No valid latency, maybe there is no AMM result?");
93  return 0;
94  }
95  std::sort(validLatency.begin(), validLatency.end());
96  double t = nonZeroCnt;
97  t = t * fraction;
98  size_t idx = (size_t) t + 1;
99  if (idx >= validLatency.size()) {
100  idx = validLatency.size() - 1;
101  }
102  return validLatency[idx];
103  }

152  static double calculateRecall(std::vector<torch::Tensor> groundTruth, std::vector<torch::Tensor> prob) {
153  int64_t truePositives = 0;
154  int64_t falseNegatives = 0;
155  for (size_t i = 0; i < prob.size(); i++) {
156  auto gdI = groundTruth[i];
157  auto probI = prob[i];
158  for (int64_t j = 0; j < probI.size(0); j++) {
159  if (existRow(gdI, probI[j])) {
160  truePositives++;
161  } else {
162  falseNegatives++;
163  }
164  }
165  }
166  double recall = static_cast<double>(truePositives) / (truePositives + falseNegatives);
167  return recall;
168  }
