260  PerfTool(pid_t pid, int cpu) {
261  if (pid == -1) { pid = gettid(); }
262  myPid = pid;
263  myCpu = cpu;
264  int nr_counters = 32;
265  for (int i = 0; i < nr_counters; i++) {
266  PerfEntry entry;
267  default_attrs[i].size = sizeof(struct perf_event_attr);
268  entry.fds = sys_perf_event_open(&default_attrs[i], pid, cpu, -1, 0);
269  if (entry.fds < 0) {
270  entry.addressable = false;
271  } else {
272  entry.addressable = true;
273  ioctl(entry.fds, PERF_EVENT_IOC_DISABLE);
274  ioctl(entry.fds, PERF_EVENT_IOC_RESET);
275  }
276  entries.push_back(entry);
277  }
278  }
279 
280  ~PerfTool() {
281  for (size_t i = 0; i < entries.size(); i++) {
282  if (entries[i].addressable == true) {
283  close(entries[i].fds);
284  //printf("close perf %d\r\n",i);
285  }
286  }
287  }

289  // reading result from a perf trace on [ch], will return 0 if the channel is invaild
290  uint64_t readPerf(size_t ch) {
291  if (ch > entries.size()) {
292  return 0;
293  }
294  if (entries[ch].addressable == false) {
295  return 0;
296  }
297  uint64_t value;
298  int ru = read(entries[ch].fds, &value, sizeof(uint64_t));
299  if (ru < 0) {
300  PERF_ERROR("invalid read");
301  }
302  return value;
303  }
