124  typedef faiss::CMax<float, VertexPtr> HC;
125  explicit MinimaxHeap(int n) : n(n), k(0), nvalid(0), ids(n), dis(n) {}
126  void push(VertexPtr i, float v) {
127  if (k == n) {
128  if (v >= dis[0]) {
129  return;
130  }
131  if (ids[0] != nullptr) {
132  --nvalid;
133  }
134  faiss::heap_pop<HC>(k--, dis.data(), ids.data());
135  }
136  faiss::heap_push<HC>(++k, dis.data(), ids.data(), v, i);
137  ++nvalid;
138  };
139  float max() const { return dis[0]; };
140  int size() const { return nvalid; };
141  void clear() {
142  nvalid = 0;
143  k = 0;
144  };
145  VertexPtr pop_min(float *vmin_out = nullptr) {
146  assert(k > 0);
147  // returns min. This is an O(n) operation
148  int i = k - 1;
149  while (i >= 0) {
150  if (ids[i] != nullptr) {
151  break;
152  }
153  i--;
154  }
155  if (i == -1) {
156  return nullptr;
157  }
158  int imin = i;
159  float vmin = dis[i];
160  i--;
161  while (i >= 0) {
162  if (ids[i] != nullptr && dis[i] < vmin) {
163  vmin = dis[i];
164  imin = i;
165  }
166  i--;
167  }
168  if (vmin_out) {
169  *vmin_out = vmin;
170  }
171  auto ret = ids[imin];
172  ids[imin] = nullptr;
173  --nvalid;
174  return ret;
175  };
176  int count_below(float thresh) {
177  int n_below = 0;
178  for (int i = 0; i < k; i++) {
179  if (dis[i] < thresh) {
180  n_below++;
181  }
182  }
183  return n_below;
184  }
185  };
