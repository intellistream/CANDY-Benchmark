137 class PQEncoder {
138  public:
139  uint8_t *code_;
141  int64_t nbits_;
142  uint8_t offset_;
143  uint8_t reg_;
144 
145  inline PQEncoder(uint8_t *code, int64_t nbits, uint8_t offset)
146  : code_(code), nbits_(nbits), offset_(offset), reg_(0) {
147  assert(nbits <= 64);
148  if (offset_ > 0) {
149  reg_ = (*code_ & ((1 << offset_) - 1));
150  }
151  };
156  inline void encode(uint64_t x) {
157  reg_ = reg_ | (uint8_t) (x << offset_);
158  x = x >> (8 - offset_);
159  if (offset_ + nbits_ >= 8) {
160  *code_++ = reg_;
161 
162  for (int i = 0; i < (nbits_ - (8 - offset_)) / 8; i++) {
163  *code_++ = (uint8_t) x;
164  x = x >> 8;
165  }
166 
167  offset_ += nbits_;
168  offset_ &= 7;
169  reg_ = (uint8_t) x;
170  } else {
171  offset_ += nbits_;
172  }
173 
174  }
175 
176  inline ~PQEncoder() {
177  if (offset_ > 0) {
178  *code_ = reg_;
179  }
180  };
181 
182 };

188 class PQDecoder {
189  public:
190  const uint8_t *code_;
191  uint8_t offset_;
192  const int64_t nbits_;
193  const uint64_t mask_;
194  uint8_t reg_;
195  inline PQDecoder(const uint8_t *code, int64_t nbits)
196  : code_(code), offset_(0), nbits_(nbits), mask_((1ull << nbits) - 1), reg_(0) {
197  assert(nbits <= 64);
198  };
203  inline uint64_t decode() {
204  if (offset_ == 0) {
205  reg_ = *code_;
206  }
207  uint64_t c = (reg_ >> offset_);
208 
209  if (offset_ + nbits_ >= 8) {
210  uint64_t e = 8 - offset_;
211  ++code_;
212  for (int i = 0; i < (nbits_ - (8 - offset_)) / 8; ++i) {
213  c = c | ((uint64_t) (*code_++) << e);
214  e += 8;
215  }
216 
217  offset_ += nbits_;
218  offset_ = offset_ & 7;
219  if (offset_ > 0) {
220  reg_ = *code_;
221  c = c | ((uint64_t) reg_ << e);
222  }
223  } else {
224  offset_ += nbits_;
225  }
226  return c & mask_;
227  };
228 };
