121  float lvq_first_level(const float* x, const size_t len, int8_t* codes){
122  assert(mean_);
123  size_t min_index =0;
124  size_t max_index =0;
125  float min = x[min_index] - (*mean_)[min_index];
126  float max = x[max_index] - (*mean_)[max_index];
127 
128  for(size_t i=1; i<len; i++){
129  if(x[i]-(*mean_)[i]<min){
130  min_index = i;
131  min = x[i] - (*mean_)[i];
132  } else if(x[i] - (*mean_)[i]>max){
133  max_index = i;
134  max = x[i] - (*mean_)[i];
135  }
136  }
137 
138  float u = max;
139  float l = min;
140 
141  float delta = (u-l)/(pow(2.0, 8)-1);
142  for(size_t i=0; i<len; i++){
143  float temp = (delta * (std::floor((x[i]-(*mean_)[i]-l)/delta)+0.5) +l);
144  codes[i] = (int)(0xff*temp);
145  }
146  return delta;
147 
148  }
