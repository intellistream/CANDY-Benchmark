108  vector<tsType> genZipfInt(size_t len, tsType maxV, double fac) {
109  vector<tsType> ret(len);
110  vector<tsType> alphabet = genIncrementalAlphabet<tsType>(maxV);
111  std::mt19937_64 gen;
112  if (!hasSeed) {
113  gen = std::mt19937_64(rd()); // 以 rd() 播种的标准 mersenne_twister_engine
114  } else {
115  gen = std::mt19937_64(seed);
116  seed++;
117  }
118 
119  std::uniform_real_distribution<> dis(0, 1);
120  vector<double> lut = genZipfLut<double>(maxV, fac);
121  for (size_t i = 0; i < len; i++) {
122  /* take random number */
123  double r = dis(gen);
124  /* binary search in lookup table to determine item */
125  size_t left = 0;
126  size_t right = maxV - 1;
127  size_t m; /* middle between left and right */
128  size_t pos; /* position to take */
129 
130  if (lut[0] >= r)
131  pos = 0;
132  else {
133  while (right - left > 1) {
134  m = (left + right) / 2;
135 
136  if (lut[m] < r)
137  left = m;
138  else
139  right = m;
140  }
141 
142  pos = right;
143  }
144  ret[i] = alphabet[pos];
145  }
146  return ret;
147  }
