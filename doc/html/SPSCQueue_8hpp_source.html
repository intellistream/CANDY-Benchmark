44  explicit SPSCQueue(const size_t capacity,
45  const Allocator &allocator = Allocator())
46  : capacity_(capacity), allocator_(allocator) {
47 
48  // The queue needs at least one element
49  if (capacity_ < 1) {
50  capacity_ = 1;
51  }
52  capacity_++; // Needs one slack element
53  // Prevent overflowing size_t
54  if (capacity_ > SIZE_MAX - 2 * kPadding) {
55  capacity_ = SIZE_MAX - 2 * kPadding;
56  }
57 
58 #if defined(__cpp_if_constexpr) && defined(__cpp_lib_void_t)
59  if constexpr (has_allocate_at_least<Allocator>::value) {
60  auto res = allocator_.allocate_at_least(capacity_ + 2 * kPadding);
61  slots_ = res.ptr;
62  capacity_ = res.count - 2 * kPadding;
63  } else {
64  slots_ = std::allocator_traits<Allocator>::allocate(
65  allocator_, capacity_ + 2 * kPadding);
66  }
67 #else

69  allocator_, capacity_ + 2 * kPadding);

117  template<typename... Args>
118  void emplace(Args &&...args)
119  noexcept(
120  std::is_nothrow_constructible<T, Args && ...>::value) {
121  static_assert(std::is_constructible<T, Args &&...>::value,
122  "T must be constructible with Args&&...");
123  auto const writeIdx = writeIdx_.load(std::memory_order_relaxed);
124  auto nextWriteIdx = writeIdx + 1;
125  if (nextWriteIdx == capacity_) {
126  nextWriteIdx = 0;
127  }
128  while (nextWriteIdx == readIdxCache_) {
129  readIdxCache_ = readIdx_.load(std::memory_order_acquire);
130  }
131  new(&slots_[writeIdx + kPadding]) T(std::forward<Args>(args)...);
132  writeIdx_.store(nextWriteIdx, std::memory_order_release);
133  }

187  T *front()
188  noexcept {
189  auto const readIdx = readIdx_.load(std::memory_order_relaxed);
190  if (readIdx == writeIdxCache_) {
191  writeIdxCache_ = writeIdx_.load(std::memory_order_acquire);
192  if (writeIdxCache_ == readIdx) {
193  return nullptr;
194  }
195  }
196  return &slots_[readIdx + kPadding];
197  }
198 
199  void pop()
200  noexcept {
201  static_assert(std::is_nothrow_destructible<T>::value,
202  "T must be nothrow destructible");
203  auto const readIdx = readIdx_.load(std::memory_order_relaxed);
204  assert(writeIdx_.load(std::memory_order_acquire) != readIdx);
205  slots_[readIdx + kPadding].~T();
206  auto nextReadIdx = readIdx + 1;
207  if (nextReadIdx == capacity_) {
208  nextReadIdx = 0;
209  }
210  readIdx_.store(nextReadIdx, std::memory_order_release);
211  }
